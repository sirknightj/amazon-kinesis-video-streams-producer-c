cmake_minimum_required(VERSION 3.6.3)

project(libopenssl-download LANGUAGES C)

if (WIN32)
  find_program(NMAKE_EXE NAMES nmake)
  find_program(MSBUILD_EXE NAMES msbuild)

  if (NMAKE_EXE)
    set(BUILD_COMMAND ${NMAKE_EXE})
    set(INSTALL_COMMAND ${NMAKE_EXE} install_sw)
  elseif (MSBUILD_EXE)
    set(BUILD_COMMAND ${MSBUILD_EXE} ${CMAKE_CURRENT_BINARY_DIR}/build/src/project_libopenssl/openssl.sln /p:Configuration=Release /p:Platform=${CMAKE_SYSTEM_PROCESSOR})
    set(INSTALL_COMMAND ${MSBUILD_EXE} ${CMAKE_CURRENT_BINARY_DIR}/build/src/project_libopenssl/openssl.sln /p:Configuration=Release /p:Platform=${CMAKE_SYSTEM_PROCESSOR} /target:INSTALL)
  else()
    message(FATAL_ERROR "No suitable build tool found (nmake or msbuild required). You probably need to invoke 'C:\Program Files\Microsoft Visual Studio\2022\Enterprise\VC\Auxiliary\Build\vcvars64.bat' to add msbuild to your PATH")
  endif()

  # Determine the target architecture
  if (CMAKE_SYSTEM_PROCESSOR MATCHES "AMD64|x86_64")
    set(OPENSSL_TARGET "VC-WIN64A")
  elseif (CMAKE_SYSTEM_PROCESSOR MATCHES "ARM64|aarch64")
    set(OPENSSL_TARGET "VC-WIN64-ARM")
  elseif (CMAKE_SYSTEM_PROCESSOR MATCHES "x86")
    set(OPENSSL_TARGET "VC-WIN32")
  else()
    message(FATAL_ERROR "Unsupported architecture: ${CMAKE_SYSTEM_PROCESSOR}")
  endif()

  SET(CONFIGURE_COMMAND perl ${CMAKE_CURRENT_BINARY_DIR}/build/src/project_libopenssl/Configure ${OPENSSL_TARGET} no-asm --prefix=${OPEN_SRC_INSTALL_PREFIX} --openssldir=${OPEN_SRC_INSTALL_PREFIX})
else()
  find_program(MAKE_EXE NAMES make)
  set(BUILD_COMMAND ${MAKE_EXE} -j)
  set(INSTALL_COMMAND ${MAKE_EXE} install_sw)

  if (BUILD_STATIC)
    SET(OPENSSL_EXTRA ${OPENSSL_EXTRA} no-shared no-dso)
  endif()

  if (DEFINED CMAKE_OSX_SYSROOT AND NOT CMAKE_OSX_SYSROOT STREQUAL "")
      if ("${CMAKE_C_COMPILER_ID}" MATCHES "Clang")
        SET(OPENSSL_EXTRA ${OPENSSL_EXTRA} -I${CMAKE_OSX_SYSROOT}/usr/include -L${CMAKE_OSX_SYSROOT}/usr/lib)
      endif()
  endif()

  if (DEFINED BUILD_OPENSSL_PLATFORM AND NOT BUILD_OPENSSL_PLATFORM STREQUAL OFF)
    SET(CONFIGURE_COMMAND ${CMAKE_CURRENT_BINARY_DIR}/build/src/project_libopenssl/Configure ${OPENSSL_EXTRA} --prefix=${OPEN_SRC_INSTALL_PREFIX} --openssldir=${OPEN_SRC_INSTALL_PREFIX} ${BUILD_OPENSSL_PLATFORM} -Wno-nullability-completeness -Wno-expansion-to-defined)
  else()
    SET(CONFIGURE_COMMAND ${CMAKE_CURRENT_BINARY_DIR}/build/src/project_libopenssl/config ${OPENSSL_EXTRA} --prefix=${OPEN_SRC_INSTALL_PREFIX} --openssldir=${OPEN_SRC_INSTALL_PREFIX} -Wno-nullability-completeness -Wno-expansion-to-defined)
  endif()
endif()

include(ExternalProject)
ExternalProject_Add(project_libopenssl
    GIT_REPOSITORY    https://github.com/openssl/openssl.git
    GIT_TAG           OpenSSL_1_1_1t
    PREFIX            ${CMAKE_CURRENT_BINARY_DIR}/build
    CONFIGURE_COMMAND ${CONFIGURE_COMMAND}
    BUILD_COMMAND     ${BUILD_COMMAND}
    BUILD_IN_SOURCE   TRUE
    INSTALL_COMMAND   ${INSTALL_COMMAND}
    TEST_COMMAND      ""
)
